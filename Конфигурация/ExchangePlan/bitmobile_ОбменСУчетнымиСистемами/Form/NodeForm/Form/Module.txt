
&AtServer
Procedure ЗагрузитьПравилаAtServer(ФайлПравил)
	
	Obj = FormAttributeToValue("Object");
	Obj.ЗагрузитьПравилаAtServerObj(ФайлПравил);
		
EndProcedure

&AtClient
Procedure ЗагрузитьПравила(Command)
	
	Если ЗначениеЗаполнено(Object.ПутьКФайлуПравил) Тогда	
		
		ФайлПравил = Новый ДвоичныеДанные(Object.ПутьКФайлуПравил);
		ЗагрузитьПравилаAtServer(ФайлПравил);
		
	КонецЕсли;		
	
EndProcedure

&AtClient
Procedure ПутьКФайлуПравилStartChoice(Item, ChoiceData, StandardProcessing)
	
	ДиалогФыбораФайла = Новый ДиалогВыбораФайла(РежимДиалогаВыбораФайла.Открытие);
	ДиалогФыбораФайла.Фильтр = "Файл xml (*.xml)|*.xml"; 
	ДиалогФыбораФайла.Заголовок = "Выберите файл";
	ДиалогФыбораФайла.ПредварительныйПросмотр = Истина;
	ДиалогФыбораФайла.ИндексФильтра = 0;
	
	Если ДиалогФыбораФайла.Выбрать() Тогда
		
		Object.ПутьКФайлуПравил = ДиалогФыбораФайла.ПолноеИмяФайла;	
		
	КонецЕсли;
	
EndProcedure
