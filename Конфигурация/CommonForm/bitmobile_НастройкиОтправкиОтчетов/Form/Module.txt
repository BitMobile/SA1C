&НаСервере
Функция ПолучениеКонстантСервер(ИмяКонстанты) 
	
	Возврат Константы[ИмяКонстанты].Получить();
	
КонецФункции

&НаСервере
Процедура ЗаписьКонстантСервер(ИмяКонстанты, ЗначениеКонстанты)
	
	Константы[ИмяКонстанты].Установить(ЗначениеКонстанты);
	
КонецПроцедуры

// Командный интерфейс

&НаКлиенте
Процедура ПриОткрытии(Отказ)
	
	КонстПрикреплятьФайлДанных = ПолучениеКонстантСервер("bitmobile_ПрикреплятьФайлДанных");
	
	Если КонстПрикреплятьФайлДанных Тогда 
		Элементы.bitmobile_РазмерПрикрепляемогоФайла.Видимость = Истина;
	Иначе
		Элементы.bitmobile_РазмерПрикрепляемогоФайла.Видимость = Ложь;
	КонецЕсли;
	
КонецПроцедуры

&НаКлиенте
Процедура ПередЗаписью(Отказ, ПараметрыЗаписи)
	
	ЗаписьКонстантСервер("bitmobile_ПрикреплятьФайлДанных", КонстПрикреплятьФайлДанных);
	
КонецПроцедуры

&НаКлиенте
Процедура ПрикреплятьФайлДанныхПриИзменении(Элемент)
	
	Если КонстПрикреплятьФайлДанных Тогда 
		Элементы.bitmobile_РазмерПрикрепляемогоФайла.Видимость = Истина;
	Иначе
		Элементы.bitmobile_РазмерПрикрепляемогоФайла.Видимость = Ложь;
	КонецЕсли;
	
КонецПроцедуры
