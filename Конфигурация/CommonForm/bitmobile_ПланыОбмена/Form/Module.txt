
&НаСервере
Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	
	// Заполнить таблицу планов обмена
	Для Каждого ПланОбмена Из Метаданные.ПланыОбмена Цикл
		
		Вставка				= ТаблицаПлановОбмена.Добавить();
		Вставка.ПланИмя		= ПланОбмена.Имя;
		Вставка.ПланСиноним	= ПланОбмена.Синоним;
		
	КонецЦикла;
	
КонецПроцедуры

&НаКлиенте
Процедура ТаблицаПлановОбменаВыбор(Элемент, ВыбраннаяСтрока, Поле, СтандартнаяОбработка)
	
	Если Не Элементы.ТаблицаПлановОбмена.ТекущиеДанные = Неопределено Тогда
		
		Если ЗначениеЗаполнено(Элементы.ТаблицаПлановОбмена.ТекущиеДанные.ПланИмя) Тогда
			
			ОткрытьФорму("ExchangePlan." + Элементы.ТаблицаПлановОбмена.ТекущиеДанные.ПланИмя + ".ФормаСписка");
			
		КонецЕсли;
		
	КонецЕсли;
	
КонецПроцедуры
